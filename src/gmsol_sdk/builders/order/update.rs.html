70/// Hint for [`UpdateOrder`].
71#[cfg_attr(js, derive(tsify_next::Tsify))]
72#[cfg_attr(js, tsify(from_wasm_abi))]
73#[cfg_attr(serde, derive(serde::Serialize, serde::Deserialize))]
74#[derive(Debug, Clone, TypedBuilder)]
75pub struct UpdateOrderHint {
76    /// Market token.
77    #[builder(setter(into))]
78    pub market_token: StringPubkey,
79}
80
81impl IntoAtomicGroup for UpdateOrder {
82    type Hint = UpdateOrderHint;
83
84    fn into_atomic_group(self, hint: &Self::Hint) -> gmsol_solana_utils::Result<AtomicGroup> {
85        let payer = self.payer.0;
86        let update = self
87            .program
88            .instruction(args::UpdateOrder {
89                params: self.params.into(),
90            })
91            .accounts(
92                accounts::UpdateOrder {
93                    owner: payer,
94                    store: self.program.store.0,
95                    market: self.program.find_market_address(&hint.market_token),
96                    order: self.order.0,
97                },
98                true,
99            )
100            .build();
101        Ok(AtomicGroup::with_instructions(&payer, Some(update)))
102    }
103}
