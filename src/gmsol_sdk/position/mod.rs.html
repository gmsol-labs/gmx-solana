13/// Position Calculations.
14pub trait PositionCalculations {
15    /// Calculate position status.
16    fn status(&self, prices: &Prices<u128>) -> crate::Result<PositionStatus>;
17}
18
19impl PositionCalculations for PositionModel {
20    fn status(&self, prices: &Prices<u128>) -> crate::Result<PositionStatus> {
21        // collateral value
22        let collateral_value = self.collateral_value(prices)?;
23
24        // pnl
25        let position_size_in_tokens = self.size_in_tokens();
26        let position_size_in_usd = self.size_in_usd();
27        let _position_size_in_usd_real = position_size_in_tokens
28            .checked_mul(prices.index_token_price.max)
29            .ok_or(gmsol_model::Error::Computation(
30                "calculating position size in usd real",
31            ))?;
32        let (pending_pnl_value, _uncapped_pnl_value, _size_delta_in_tokens) =
33            self.pnl_value(prices, position_size_in_usd)?;
34        let entry_price = position_size_in_usd
35            .checked_div(*position_size_in_tokens)
36            .ok_or(gmsol_model::Error::Computation("calculating entry price"))?;
37
38        // borrowing fee value
39        let pending_borrowing_fee_value = self.pending_borrowing_fee_value()?;
40
41        // funding fee value
42        let pending_funding_fee = self.pending_funding_fees()?;
43        let pending_funding_fee_value = if self.is_collateral_token_long() {
44            pending_funding_fee
45                .amount()
46                .checked_mul(prices.long_token_price.min)
47                .ok_or(gmsol_model::Error::Computation(
48                    "calculating pending funding fee value",
49                ))?
50        } else {
51            pending_funding_fee
52                .amount()
53                .checked_mul(prices.short_token_price.min)
54                .ok_or(gmsol_model::Error::Computation(
55                    "calculating pending funding fee value",
56                ))?
57        };
58        let pending_claimable_funding_fee_value_in_long_token = pending_funding_fee
59            .claimable_long_token_amount()
60            .checked_mul(prices.long_token_price.min)
61            .ok_or(gmsol_model::Error::Computation(
62                "calculating pending claimable funding fee value in long token",
63            ))?;
64        let pending_claimable_funding_fee_value_in_short_token = pending_funding_fee
65            .claimable_short_token_amount()
66            .checked_mul(prices.short_token_price.min)
67            .ok_or(gmsol_model::Error::Computation(
68                "calculating pending claimable funding fee value in short token",
69            ))?;
70
71        // close order fee value
72        let collateral_token_price = if self.is_collateral_token_long() {
73            prices.long_token_price
74        } else {
75            prices.short_token_price
76        };
77
78        // net value = collateral value +  pending pnl - pending borrowing fee value - nagetive pending funding fee value - close order fee value let mut price_impact_value = self.position_price_impact(&size_delta_usd)?;
79        let size_delta_usd = position_size_in_usd.to_opposite_signed()?;
80        let mut price_impact_value = self.position_price_impact(&size_delta_usd)?;
81        let has_positive_impact = price_impact_value.is_positive();
82        if price_impact_value.is_negative() {
83            self.market().cap_negative_position_price_impact(
84                &size_delta_usd,
85                true,
86                &mut price_impact_value,
87            )?;
88        } else {
89            price_impact_value = Zero::zero();
90        }
91
92        let total_position_fees = self.position_fees(
93            &collateral_token_price,
94            position_size_in_usd,
95            has_positive_impact,
96            // Should not account for liquidation fees to determine if position should be liquidated.
97            false,
98        )?;
99
100        let close_order_fee_value = *total_position_fees.order_fees().fee_value();
101
102        let net_value = collateral_value
103            .to_signed()?
104            .checked_add(pending_pnl_value)
105            .ok_or(gmsol_model::Error::Computation("calculating net value"))?
106            .checked_sub(pending_borrowing_fee_value.to_signed()?)
107            .ok_or(gmsol_model::Error::Computation("calculating net value"))?
108            .checked_sub(pending_funding_fee_value.to_signed()?)
109            .ok_or(gmsol_model::Error::Computation("calculating net value"))?
110            .checked_sub(close_order_fee_value.to_signed()?)
111            .ok_or(gmsol_model::Error::Computation("calculating net value"))?
112            .max(Zero::zero());
113
114        // leverage
115        let leverage = if !net_value.is_positive() {
116            None
117        } else {
118            Some(
119                gmsol_model::utils::div_to_factor::<_, { constants::MARKET_DECIMALS }>(
120                    position_size_in_usd,
121                    &net_value.unsigned_abs(),
122                    true,
123                )
124                .ok_or(gmsol_model::Error::Computation("calculating leverage"))?,
125            )
126        };
127
128        // liquidation price
129        let params = self.market().position_params()?;
130        let min_collateral_factor = params.min_collateral_factor();
131        let min_collateral_value = params.min_collateral_value();
132        let liquidation_collateral_usd = gmsol_model::utils::apply_factor::<
133            _,
134            { constants::MARKET_DECIMALS },
135        >(position_size_in_usd, min_collateral_factor)
136        .max(Some(*min_collateral_value))
137        .ok_or(gmsol_model::Error::Computation(
138            "calculating liquidation collateral usd",
139        ))?;
140
141        let liquidation_price = if position_size_in_tokens.is_zero() {
142            None
143        } else {
144            collateral_value
145                .checked_add_signed(price_impact_value)
146                .and_then(|a| a.checked_sub(pending_borrowing_fee_value))
147                .and_then(|a| a.checked_sub(pending_funding_fee_value))
148                .and_then(|a| a.checked_sub(close_order_fee_value))
149                .and_then(|remaining_collateral_usd| {
150                    if self.is_long() {
151                        liquidation_collateral_usd
152                            .checked_add(*position_size_in_usd)?
153                            .checked_sub(remaining_collateral_usd)?
154                            .checked_div(*position_size_in_tokens)
155                    } else {
156                        remaining_collateral_usd
157                            .checked_add(*position_size_in_usd)?
158                            .checked_sub(liquidation_collateral_usd)?
159                            .checked_div(*position_size_in_tokens)
160                    }
161                })
162        };
163
164        Ok(PositionStatus {
165            entry_price,
166            collateral_value,
167            pending_pnl: pending_pnl_value,
168            pending_borrowing_fee_value,
169            pending_funding_fee_value,
170            pending_claimable_funding_fee_value_in_long_token,
171            pending_claimable_funding_fee_value_in_short_token,
172            close_order_fee_value,
173            net_value,
174            leverage,
175            liquidation_price,
176        })
177    }
178}
