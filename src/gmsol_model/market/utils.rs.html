7/// Utils for market.
8pub(crate) trait MarketUtils<const DECIMALS: u8>: BaseMarket<DECIMALS> {
9    fn usage_factor(
10        &self,
11        is_long: bool,
12        reserved_value: &Self::Num,
13        pool_value: &Self::Num,
14    ) -> crate::Result<Self::Num> {
15        let reserve_factor = self.open_interest_reserve_factor()?;
16        let max_reserved_value = utils::apply_factor(pool_value, &reserve_factor).ok_or(
17            crate::Error::Computation("usage factor: calculating max reserved factor"),
18        )?;
19        let reserve_usage_factor = utils::div_to_factor(reserved_value, &max_reserved_value, false)
20            .ok_or(crate::Error::Computation(
21                "usage factor: calculating reserve usage factor",
22            ))?;
23
24        if self.ignore_open_interest_for_usage_factor()? {
25            return Ok(reserve_usage_factor);
26        }
27
28        let max_open_interest = self.max_open_interest(is_long)?;
29        let open_interest = self.open_interest()?.amount(is_long)?;
30        let open_interest_usage_factor =
31            utils::div_to_factor(&open_interest, &max_open_interest, false).ok_or(
32                crate::Error::Computation("usage factor: calculating open interest usage factor"),
33            )?;
34
35        if reserve_usage_factor > open_interest_usage_factor {
36            Ok(reserve_usage_factor)
37        } else {
38            Ok(open_interest_usage_factor)
39        }
40    }
41
42    /// Cap pnl with the given pool value and the max pnl factor.
43    fn cap_pnl(
44        &self,
45        is_long: bool,
46        pnl: &Self::Signed,
47        pool_value: &Self::Num,
48        kind: PnlFactorKind,
49    ) -> crate::Result<Self::Signed> {
50        if pnl.is_positive() {
51            let max_pnl_factor = self.pnl_factor_config(kind, is_long)?;
52            let max_pnl = crate::utils::apply_factor(pool_value, &max_pnl_factor)
53                .ok_or(crate::Error::Computation("calculating max pnl"))?
54                .to_signed()?;
55            if *pnl > max_pnl {
56                Ok(max_pnl)
57            } else {
58                Ok(pnl.clone())
59            }
60        } else {
61            Ok(pnl.clone())
62        }
63    }
64}
