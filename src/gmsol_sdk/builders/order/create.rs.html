18/// Create Order Kind.
19#[cfg_attr(js, derive(tsify_next::Tsify))]
20#[cfg_attr(js, tsify(from_wasm_abi))]
21#[cfg_attr(serde, derive(serde::Serialize, serde::Deserialize))]
22#[derive(Debug, Clone, Copy)]
23pub enum CreateOrderKind {
24    /// Market Swap.
25    MarketSwap,
26    /// Market Increase.
27    MarketIncrease,
28    /// Market Decrease.
29    MarketDecrease,
30    /// Limit Swap.
31    LimitSwap,
32    /// Limit Increase.
33    LimitIncrease,
34    /// Limit Decrease.
35    LimitDecrease,
36    /// Stop-loss Decrease.
37    StopLossDecrease,
38}
39
40impl From<CreateOrderKind> for OrderKind {
41    fn from(kind: CreateOrderKind) -> Self {
42        match kind {
43            CreateOrderKind::MarketSwap => Self::MarketSwap,
44            CreateOrderKind::MarketIncrease => Self::MarketIncrease,
45            CreateOrderKind::MarketDecrease => Self::MarketDecrease,
46            CreateOrderKind::LimitSwap => Self::LimitSwap,
47            CreateOrderKind::LimitIncrease => Self::LimitIncrease,
48            CreateOrderKind::LimitDecrease => Self::LimitDecrease,
49            CreateOrderKind::StopLossDecrease => Self::StopLossDecrease,
50        }
51    }
52}
53
54impl CreateOrderKind {
55    /// Returns whether the order kind is "swap".
56    pub fn is_swap(&self) -> bool {
57        matches!(self, Self::MarketSwap | Self::LimitSwap)
58    }
59
60    /// Returns whether the order kind is "increase".
61    pub fn is_increase(&self) -> bool {
62        matches!(self, Self::MarketIncrease | Self::LimitIncrease)
63    }
64
65    /// Returns whether the ordr kind is "decrease".
66    pub fn is_decrease(&self) -> bool {
67        matches!(
68            self,
69            Self::MarketDecrease | Self::LimitDecrease | Self::StopLossDecrease
70        )
71    }
72}

204impl CreateOrder {
205    fn is_collateral_or_swap_out_token_long(
206        &self,
207        hint: &CreateOrderHint,
208    ) -> Result<bool, crate::SolanaUtilsError> {
209        let token = &*self.collateral_or_swap_out_token;
210        if *token == *hint.long_token {
211            Ok(true)
212        } else if *token == *hint.short_token {
213            Ok(false)
214        } else {
215            Err(crate::SolanaUtilsError::custom(
216                "invalid hint: `collateral_or_swap_out_token` is not one of the specified long or short tokens",
217            ))
218        }
219    }
220}

222/// Hint for [`CreateOrder`].
223#[cfg_attr(js, derive(tsify_next::Tsify))]
224#[cfg_attr(js, tsify(from_wasm_abi))]
225#[cfg_attr(serde, derive(serde::Serialize, serde::Deserialize))]
226#[derive(Debug, Clone, TypedBuilder)]
227pub struct CreateOrderHint {
228    /// Long token.
229    #[builder(setter(into))]
230    pub long_token: StringPubkey,
231    /// Short token.
232    #[builder(setter(into))]
233    pub short_token: StringPubkey,
234}
235
236impl IntoAtomicGroup for CreateOrder {
237    type Hint = CreateOrderHint;
238
239    fn into_atomic_group(self, hint: &Self::Hint) -> Result<AtomicGroup, crate::SolanaUtilsError> {
240        let owner = self.payer.0;
241
242        let mut insts = AtomicGroup::new(&owner);
243
244        let receiver = self.receiver.as_deref().copied().unwrap_or(owner);
245        let nonce = self.nonce.unwrap_or_else(generate_nonce);
246        let order = self.program.find_order_address(&owner, &nonce);
247        let token_program_id = anchor_spl::token::ID;
248
249        let collateral_or_swap_out_token = self.collateral_or_swap_out_token.0;
250        let is_collateral_or_swap_out_token_long =
251            self.is_collateral_or_swap_out_token_long(hint)?;
252
253        let (pay_token, receive_token, long_token, short_token, is_position_order) = match self.kind
254        {
255            CreateOrderKind::MarketSwap | CreateOrderKind::LimitSwap => {
256                if let Some(receive_token) = self.receive_token {
257                    if receive_token.0 != collateral_or_swap_out_token {
258                        return Err(crate::SolanaUtilsError::custom("invalid `receive_token`: it must be the same as `collateral_or_swap_out_token` for swap orders if provided"));
259                    }
260                }
261                (
262                    Some(
263                        self.pay_token
264                            .as_deref()
265                            .copied()
266                            .unwrap_or(collateral_or_swap_out_token),
267                    ),
268                    Some(collateral_or_swap_out_token),
269                    None,
270                    None,
271                    false,
272                )
273            }
274            CreateOrderKind::MarketIncrease | CreateOrderKind::LimitIncrease => (
275                Some(
276                    self.pay_token
277                        .as_deref()
278                        .copied()
279                        .unwrap_or(collateral_or_swap_out_token),
280                ),
281                None,
282                Some(hint.long_token.0),
283                Some(hint.short_token.0),
284                true,
285            ),
286            CreateOrderKind::MarketDecrease
287            | CreateOrderKind::LimitDecrease
288            | CreateOrderKind::StopLossDecrease => (
289                None,
290                Some(
291                    self.receive_token
292                        .as_deref()
293                        .copied()
294                        .unwrap_or(collateral_or_swap_out_token),
295                ),
296                Some(hint.long_token.0),
297                Some(hint.short_token.0),
298                true,
299            ),
300        };
301
302        let pay_token_account = pay_token.as_ref().map(|token| {
303            self.pay_token_account
304                .as_deref()
305                .copied()
306                .unwrap_or_else(|| {
307                    get_associated_token_address_with_program_id(&owner, token, &token_program_id)
308                })
309        });
310        let (pay_token_escrow, prepare) =
311            prepare_ata(&owner, &order, pay_token.as_ref(), &token_program_id).unzip();
312        insts.extend(prepare);
313        let (receive_token_escrow, prepare) =
314            prepare_ata(&owner, &order, receive_token.as_ref(), &token_program_id).unzip();
315        insts.extend(prepare);
316        let (long_token_escrow, prepare) =
317            prepare_ata(&owner, &order, long_token.as_ref(), &token_program_id).unzip();
318        insts.extend(prepare);
319        let (short_token_escrow, prepare) =
320            prepare_ata(&owner, &order, short_token.as_ref(), &token_program_id).unzip();
321        insts.extend(prepare);
322
323        let market = self.program.find_market_address(&self.params.market_token);
324        let user = self.program.find_user_address(&owner);
325        let position = (is_position_order).then(|| {
326            self.program.find_position_address(
327                &owner,
328                &self.params.market_token,
329                &collateral_or_swap_out_token,
330                self.params.is_long,
331            )
332        });
333        let params = &self.params;
334        let swap_markets = self
335            .swap_path
336            .iter()
337            .map(|mint| AccountMeta {
338                pubkey: self.program.find_market_address(mint),
339                is_signer: false,
340                is_writable: false,
341            })
342            .collect::<Vec<_>>();
343
344        let params = StoreCreateOrderParams {
345            kind: self.kind.into(),
346            decrease_position_swap_type: params.decrease_position_swap_type.map(Into::into),
347            execution_lamports: self.execution_lamports,
348            swap_path_length: self.swap_path.len() as u8,
349            initial_collateral_delta_amount: self
350                .params
351                .amount
352                .try_into()
353                .map_err(crate::SolanaUtilsError::custom)?,
354            size_delta_value: self.params.size,
355            is_long: self.params.is_long,
356            is_collateral_long: is_collateral_or_swap_out_token_long,
357            min_output: Some(self.params.min_output),
358            trigger_price: self.params.trigger_price,
359            acceptable_price: self.params.acceptable_price,
360            should_unwrap_native_token: self.unwrap_native_on_receive,
361            valid_from_ts: self.params.valid_from_ts,
362        };
363
364        if self.kind.is_increase() {
365            insts.add(
366                self.program
367                    .instruction(args::PreparePosition { params })
368                    .accounts(
369                        accounts::PreparePosition {
370                            owner,
371                            store: self.program.store.0,
372                            market,
373                            position: position.unwrap(),
374                            system_program: system_program::ID,
375                        },
376                        true,
377                    )
378                    .build(),
379            );
380        }
381
382        let create = self
383            .program
384            .instruction(args::CreateOrder {
385                nonce: nonce.to_bytes(),
386                params,
387            })
388            .accounts(
389                accounts::CreateOrder {
390                    owner,
391                    receiver,
392                    store: self.program.store.0,
393                    market,
394                    user,
395                    order,
396                    position,
397                    initial_collateral_token: pay_token,
398                    final_output_token: receive_token.unwrap_or(collateral_or_swap_out_token),
399                    long_token,
400                    short_token,
401                    initial_collateral_token_escrow: pay_token_escrow,
402                    final_output_token_escrow: receive_token_escrow,
403                    long_token_escrow,
404                    short_token_escrow,
405                    initial_collateral_token_source: pay_token_account,
406                    system_program: system_program::ID,
407                    token_program: token_program_id,
408                    associated_token_program: associated_token::ID,
409                },
410                true,
411            )
412            .accounts(swap_markets, false)
413            .build();
414
415        insts.add(create);
416
417        Ok(insts)
418    }
419}
