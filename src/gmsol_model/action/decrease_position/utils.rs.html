1use num_traits::{CheckedAdd, CheckedDiv, CheckedSub, Signed, Zero};
2
3use crate::{
4    num::{MulDiv, UnsignedAbs},
5    price::Price,
6};
7
8pub(super) fn get_execution_price_for_decrease<T>(
9    index_price: &Price<T>,
10    size_in_usd: &T,
11    size_in_tokens: &T,
12    size_delta_usd: &T,
13    price_impact_value: &T::Signed,
14    acceptable_price: Option<&T>,
15    is_long: bool,
16) -> crate::Result<T>
17where
18    T: Clone + MulDiv + Ord + CheckedAdd + CheckedSub,
19    T::Signed: CheckedSub + Clone + Ord + UnsignedAbs + CheckedDiv,
20{
21    let mut execution_price = index_price.pick_price(!is_long).clone();
22    if !size_delta_usd.is_zero() && !size_in_tokens.is_zero() {
23        let adjusted_price_impact_value = if is_long {
24            price_impact_value.clone()
25        } else {
26            T::Signed::zero()
27                .checked_sub(price_impact_value)
28                .ok_or(crate::Error::Computation("price impact too large"))?
29        };
30
31        if adjusted_price_impact_value.is_negative()
32            && adjusted_price_impact_value.unsigned_abs() > *size_delta_usd
33        {
34            return Err(crate::Error::Computation(
35                "price impact larger than order size",
36            ));
37        }
38
39        // Since the decimals of the USD value are often greater than those of `size_in_tokens`,
40        // we must compute the `adjustment` in the following sequence. Furthermore, since the
41        // `price_impact_value` is comparable in magnitude to `size_delta_usd` and is capped,
42        // overflow is unlikely in the `mul_div` part.
43        //
44        // In practice, since:
45        //
46        // size_in_usd * abs(price_impact_value) / size_delta_usd
47        //     <= size_in_usd * (max_position_impact_factor * size_delta_usd / UNIT) / size_delta_usd
48        //     == size_in_usd * max_position_impact_factor / UNIT
49        //
50        // To ensure that no overflow occurs, we need:
51        //
52        // Unsigned::MAX * max_position_impact_factor / UNIT <= Signed::MAX
53        //
54        // Which simplifies to:
55        //
56        // max_position_impact_factor <= (Signed::MAX / Unsigned::MAX) * UNIT
57        //
58        // In other words, as long as `max_position_impact_factor` doesn't exceed 50%.
59        //
60        // Otherwise, if we use the order `size_in_usd.mul_div(adjusted_price_impact_value, size_in_tokens) / size_in_usd`,
61        // the `mul_div` operation is likely to overflow. For example (Unsigned = u128, Signed = i128, DECIMALS = 20):
62        //
63        // Assume that `size_in_usd = 6250 * 10^20` ($6250) and `size_in_tokens = 67774` (0.067774 BTC, decimals = 6). When the user
64        // close it with `size_delta_usd = 6250 * 10^20` causing a `price_impact_value = 3.90625 * 10^20` (the "factor" is only
65        // 0.0625%), an overflow will occur in the `mul_div` step:
66        //
67        // size_in_usd * price_impact_value / size_in_tokens >= 3.60 * 10^39 > i128::MAX
68        //
69        let signed_size_in_tokens = size_in_tokens.to_signed()?;
70        let adjustment = size_in_usd
71            .checked_mul_div_with_signed_numerator(&adjusted_price_impact_value, size_delta_usd)
72            .ok_or(crate::Error::Computation(
73                "calculating execution price adjustment",
74            ))?
75            .checked_div(&signed_size_in_tokens)
76            .ok_or(crate::Error::Computation("calculating adjustment"))?;
77        execution_price = execution_price
78            .checked_add_with_signed(&adjustment)
79            .ok_or(crate::Error::Computation("adjusting execution price"))?;
80    }
81    let Some(acceptable_price) = acceptable_price else {
82        return Ok(execution_price);
83    };
84    if (is_long && execution_price >= *acceptable_price)
85        || (!is_long && execution_price <= *acceptable_price)
86    {
87        Ok(execution_price)
88    } else {
89        Err(crate::Error::InvalidArgument(
90            "order not fulfillable at acceptable price",
91        ))
92    }
93}
