118fn to_seed(key: &str) -> [u8; 32] {
119    use solana_sdk::hash::hash;
120    hash(key.as_bytes()).to_bytes()
121}

239/// Find PDA for position.
240pub fn find_position_address(
241    store: &Pubkey,
242    owner: &Pubkey,
243    market_token: &Pubkey,
244    collateral_token: &Pubkey,
245    is_long: bool,
246    store_program_id: &Pubkey,
247) -> (Pubkey, u8) {
248    // See the definition of `PositionKind`.
249    let kind = if is_long { 1 } else { 2 };
250    Pubkey::find_program_address(
251        &[
252            POSITION_SEED,
253            store.as_ref(),
254            owner.as_ref(),
255            market_token.as_ref(),
256            collateral_token.as_ref(),
257            &[kind],
258        ],
259        store_program_id,
260    )
261}
