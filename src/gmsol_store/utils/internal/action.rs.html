14/// Create Action.
15pub(crate) trait Create<'info, A>: Sized + anchor_lang::Bumps {
16    /// Create Params.
17    type CreateParams: ActionParams;
18
19    /// Get the action account.
20    fn action(&self) -> AccountInfo<'info>;
21
22    /// Get the payer account.
23    fn payer(&self) -> AccountInfo<'info>;
24
25    /// Get the seeds of the payer.
26    fn payer_seeds(&self) -> Result<Option<Vec<Vec<u8>>>> {
27        Ok(None)
28    }
29
30    /// Get the system program account.
31    fn system_program(&self) -> AccountInfo<'info>;
32
33    /// Validate.
34    fn validate(&self, _params: &Self::CreateParams) -> Result<()>;
35
36    /// The implementation of the creation.
37    fn create_impl(
38        &mut self,
39        params: &Self::CreateParams,
40        nonce: &NonceBytes,
41        bumps: &Self::Bumps,
42        remaining_accounts: &'info [AccountInfo<'info>],
43    ) -> Result<()>;
44
45    /// Create Action.
46    fn create(
47        ctx: &mut Context<'_, '_, 'info, 'info, Self>,
48        nonce: &NonceBytes,
49        params: &Self::CreateParams,
50    ) -> Result<()> {
51        let accounts = &mut ctx.accounts;
52        accounts.validate(params)?;
53        accounts.transfer_execution_lamports(params)?;
54        accounts.create_impl(params, nonce, &ctx.bumps, ctx.remaining_accounts)?;
55        Ok(())
56    }
57
58    /// Transfer execution lamports.
59    fn transfer_execution_lamports(&self, params: &Self::CreateParams) -> Result<()> {
60        use crate::ops::execution_fee::TransferExecutionFeeOperation;
61
62        let payer_seeds = self.payer_seeds()?;
63        let payer_seeds = payer_seeds
64            .as_ref()
65            .map(|seeds| seeds.iter().map(|seed| seed.as_slice()).collect::<Vec<_>>());
66
67        TransferExecutionFeeOperation::builder()
68            .payment(self.action())
69            .payer(self.payer())
70            .execution_lamports(params.execution_lamports())
71            .system_program(self.system_program())
72            .signer_seeds(payer_seeds.as_deref())
73            .build()
74            .execute()
75    }
76}

81/// Close Action.
82pub(crate) trait Close<'info, A>: Authenticate<'info>
83where
84    A: Action + ZeroCopy + Owner + Closable,
85{
86    /// Expected keeper role.
87    fn expected_keeper_role(&self) -> &str;
88
89    /// Rent receiver.
90    fn rent_receiver(&self) -> AccountInfo<'info>;
91
92    /// Get event authority.
93    fn event_authority(&self, bumps: &Self::Bumps) -> (AccountInfo<'info>, u8);
94
95    /// Get store wallet bump.
96    fn store_wallet_bump(&self, bumps: &Self::Bumps) -> u8;
97
98    /// Whether to skip the completion check when the authority is keeper.
99    fn skip_completion_check_for_keeper(&self) -> Result<bool> {
100        Ok(false)
101    }
102
103    /// Validate.
104    fn validate(&self) -> Result<()>;
105
106    /// Process before the close.
107    fn process(
108        &self,
109        init_if_needed: bool,
110        store_wallet_signer: &StoreWalletSigner,
111        event_emitter: &EventEmitter<'_, 'info>,
112    ) -> Result<Success>;
113
114    /// Close Action.
115    fn close(ctx: &Context<'_, '_, '_, 'info, Self>, reason: &str) -> Result<()> {
116        let accounts = &ctx.accounts;
117        accounts.validate()?;
118        let should_continue_when_atas_are_missing = accounts.preprocess()?;
119
120        let store_wallet_signer = StoreWalletSigner::new(
121            accounts.store().key(),
122            accounts.store_wallet_bump(&ctx.bumps),
123        );
124
125        let (authority, bump) = accounts.event_authority(&ctx.bumps);
126        let event_emitter = EventEmitter::new(&authority, bump);
127
128        if accounts.process(
129            should_continue_when_atas_are_missing,
130            &store_wallet_signer,
131            &event_emitter,
132        )? {
133            {
134                let action_address = accounts.action().key();
135                let action = accounts.action().load()?;
136                let event = action.to_closed_event(&action_address, reason)?;
137                event_emitter.emit_cpi(&event)?;
138            }
139            accounts.close_action_account()?;
140        } else {
141            msg!("Some ATAs are not initialized, skip the close");
142        }
143        Ok(())
144    }
145
146    /// Action.
147    fn action(&self) -> &AccountLoader<'info, A>;
148
149    /// Preprocess.
150    fn preprocess(&self) -> Result<ShouldContinueWhenATAsAreMissing> {
151        if *self.authority().key == self.action().load()?.header().owner {
152            Ok(true)
153        } else {
154            self.only_role(self.expected_keeper_role())?;
155            {
156                let action = self.action().load()?;
157                if self.skip_completion_check_for_keeper()?
158                    || action.header().action_state()?.is_completed_or_cancelled()
159                {
160                    Ok(false)
161                } else {
162                    err!(CoreError::PermissionDenied)
163                }
164            }
165        }
166    }
167
168    /// Close the action account.
169    fn close_action_account(&self) -> Result<()> {
170        self.action().close(self.rent_receiver())
171    }
172}
